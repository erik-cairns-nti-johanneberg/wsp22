def false_img(img)
    #check
    if img == nil
        return true

    elsif img["type"] != "image/jpeg" && img["type"] != "image/png"
        return true

    elsif get_file_size(img["tempfile"]) > MAX_IMG_SIZE
        return true

    else
        return false
    end
end
